A function to get the selected value out of a set of radio buttons:
function getSelectedRadioButton(name){
        let radios = document.getElementsByName(name)
        for(let i = 0; i < radios.length; ++i){
            if(radios[i].checked){
                return radios[i].value
            }        }
        return null
    }
